const btn = document.querySelector("#playButton"); const video = document.querySelector("#miVideo"); btn.addEventListener('click', () => { // Primero: Intentar reproducir (importante) video.play().then(() => { // Segundo: Esconder botón document.querySelector("#ui").style.display = "none"; // Tercero: Forzar Pantalla Completa según el dispositivo if (video.webkitEnterFullscreen) { video.webkitEnterFullscreen(); // Esto es clave para iPhone } else if (video.requestFullscreen) { video.requestFullscreen(); // Esto es para Android } else if (video.webkitRequestFullscreen) { video.webkitRequestFullscreen(); } }).catch(err => { console.log("Error al reproducir:", err); }); }); ```
